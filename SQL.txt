CREATE DATABASE RECORD:
CREATE DATABASE record CHARACTER SET utf8 COLLATE utf8_general_ci;

CREATE TABLE CUSTOMER:
CREATE TABLE `record`.`customer` ( `id` INT(8) NOT NULL AUTO_INCREMENT , `fname` VARCHAR(50) NOT NULL , `lname` VARCHAR(50) NOT NULL , `address` VARCHAR(150) NOT NULL , `state` VARCHAR(2) NOT NULL , `zip` INT(5) NOT NULL , `city` VARCHAR(50) NOT NULL , `phone` VARCHAR(10) NOT NULL , `email` VARCHAR(50) NOT NULL , `footage` VARCHAR(50) NOT NULL , `date_created` DATETIME on update CURRENT_TIMESTAMP NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_spanish_ci;

CREATE TABLE `customer` (
  `id` int(8) NOT NULL AUTO_INCREMENT,
  `fname` varchar(50) COLLATE utf8_spanish_ci NOT NULL,
  `lname` varchar(50) COLLATE utf8_spanish_ci NOT NULL,
  `address` varchar(150) COLLATE utf8_spanish_ci NOT NULL,
  `state` varchar(2) COLLATE utf8_spanish_ci NOT NULL,
  `zip` int(5) NOT NULL,
  `city` varchar(50) COLLATE utf8_spanish_ci NOT NULL,
  `phone` varchar(10) COLLATE utf8_spanish_ci NOT NULL,
  `email` varchar(50) COLLATE utf8_spanish_ci NOT NULL,
  `footage` varchar(50) COLLATE utf8_spanish_ci NOT NULL,
  `date_created` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_spanish_ci;
